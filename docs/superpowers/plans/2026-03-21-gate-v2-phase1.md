// src/cli/config.js
const fs = require('fs');
const path = require('path');
const yaml = require('js-yaml');

const DEFAULTS = {
  entropy_threshold: 4.0,
  verify: true,
  hooks: ['pre-commit'],
  severity: {},
  rules: [],
  output: {
    format: 'text',
    color: 'auto',
    context_lines: 2,
  },
};

/**
 * Load configuration from .gaterc (YAML) in the given directory.
 * Falls back to smart defaults if no config file or on parse error.
 *
 * @param {string} dir - Directory to look for .gaterc (default: cwd)
 * @returns {object} Merged configuration
 */
function loadConfig(dir) {
  dir = dir || process.cwd();
  const configPath = path.join(dir, '.gaterc');
  let userConfig = {};

  if (fs.existsSync(configPath)) {
    try {
      const raw = fs.readFileSync(configPath, 'utf8');
      const parsed = yaml.load(raw);
      if (parsed && typeof parsed === 'object') {
        userConfig = parsed;
      }
    } catch (err) {
      // Invalid YAML — fall back to defaults silently
      // (loadConfig is called on every scan; don't spam stderr)
    }
  }

  // Validate and compile custom rules
  const customRules = [];
  if (Array.isArray(userConfig.rules)) {
    for (const rule of userConfig.rules) {
      if (!rule.id || !rule.pattern) continue;
      try {
        // Test that the pattern compiles
        new RegExp(rule.pattern);
        customRules.push({
          id: rule.id,
          name: rule.name || rule.id,
          pattern: rule.pattern,
          severity: rule.severity || 'medium',
          remediation: rule.remediation || null,
        });
      } catch {
        // Skip rules with invalid regex
      }
    }
  }

  return {
    entropy_threshold: typeof userConfig.entropy_threshold === 'number'
      ? userConfig.entropy_threshold
      : DEFAULTS.entropy_threshold,
    verify: typeof userConfig.verify === 'boolean'
      ? userConfig.verify
      : DEFAULTS.verify,
    hooks: Array.isArray(userConfig.hooks)
      ? userConfig.hooks
      : DEFAULTS.hooks,
    severity: (userConfig.severity && typeof userConfig.severity === 'object')
      ? userConfig.severity
      : DEFAULTS.severity,
    rules: customRules,
    output: {
      format: userConfig.output?.format || DEFAULTS.output.format,
      color: userConfig.output?.color ?? DEFAULTS.output.color,
      context_lines: typeof userConfig.output?.context_lines === 'number'
        ? userConfig.output.context_lines
        : DEFAULTS.output.context_lines,
    },
  };
}

/**
 * Detect the project stack from files present.
 * @param {string} dir
 * @returns {string[]} Array of detected languages/frameworks
 */
function detectStack(dir) {
  dir = dir || process.cwd();
  const stacks = [];
  const checks = [
    { file: 'package.json', stack: 'node' },
    { file: 'requirements.txt', stack: 'python' },
    { file: 'pyproject.toml', stack: 'python' },
    { file: 'setup.py', stack: 'python' },
    { file: 'go.mod', stack: 'go' },
    { file: 'Gemfile', stack: 'ruby' },
    { file: 'Cargo.toml', stack: 'rust' },
    { file: 'pom.xml', stack: 'java' },
    { file: 'build.gradle', stack: 'java' },
    { file: 'build.gradle.kts', stack: 'java' },
  ];

  for (const { file, stack } of checks) {
    if (fs.existsSync(path.join(dir, file)) && !stacks.includes(stack)) {
      stacks.push(stack);
    }
  }

  // Check for .NET projects by extension (can't use glob in existsSync)
  if (!stacks.includes('dotnet')) {
    try {
      const files = fs.readdirSync(dir);
      if (files.some(f => f.endsWith('.sln') || f.endsWith('.csproj'))) {
        stacks.push('dotnet');
      }
    } catch { /* ignore read errors */ }
  }

  return stacks;
}

/**
 * Get default .gateignore patterns for a stack.
 * @param {string[]} stacks
 * @returns {string[]}
 */
function getDefaultIgnorePatterns(stacks) {
  const patterns = [];
  const stackPatterns = {
    node: ['node_modules/**', '*.min.js', '*.bundle.js', 'dist/**', 'build/**', 'coverage/**'],
    python: ['venv/**', '__pycache__/**', '.tox/**', '*.pyc', '.eggs/**'],
    go: ['vendor/**'],
    ruby: ['vendor/bundle/**'],
    rust: ['target/**'],
    java: ['target/**', 'build/**', '*.class'],
    dotnet: ['bin/**', 'obj/**'],
  };

  for (const stack of stacks) {
    if (stackPatterns[stack]) {
      for (const p of stackPatterns[stack]) {
        if (!patterns.includes(p)) patterns.push(p);
      }
    }
  }

  // Always ignore these
  patterns.push('.git/**');
  return patterns;
}

module.exports = { loadConfig, detectStack, getDefaultIgnorePatterns, DEFAULTS };
```

// src/cli/ignore.js
const fs = require('fs');
const path = require('path');

/**
 * Simple glob matcher supporting * and ** patterns.
 * No dependency needed — covers the patterns .gateignore uses.
 */
function globToRegex(glob) {
  let regex = '^';
  let i = 0;
  while (i < glob.length) {
    const c = glob[i];
    if (c === '*') {
      if (glob[i + 1] === '*') {
        if (glob[i + 2] === '/') {
          regex += '(?:.+/)?';
          i += 3;
        } else {
          regex += '.*';
          i += 2;
        }
      } else {
        regex += '[^/]*';
        i++;
      }
    } else if (c === '?') {
      regex += '[^/]';
      i++;
    } else if (c === '.') {
      regex += '\\.';
      i++;
    } else {
      regex += c;
      i++;
    }
  }
  regex += '$';
  return new RegExp(regex);
}

const RULE_PATTERN_RE = /^\[rule:([^\]]+)\]\s+(.+)$/;
const INLINE_IGNORE_RE = /\/[\/\*]\s*gate-ignore(?::\s*(.+?)\s*)?(?:\*\/)?$/;

/**
 * Load ignore patterns from .gateignore
 * @param {string} dir
 * @returns {{ filePatterns: RegExp[], rulePatterns: Array<{ruleId: string, glob: string, regex: RegExp}> }}
 */
function loadIgnorePatterns(dir) {
  dir = dir || process.cwd();
  const result = { filePatterns: [], rulePatterns: [] };
  const ignorePath = path.join(dir, '.gateignore');

  if (!fs.existsSync(ignorePath)) return result;

  const lines = fs.readFileSync(ignorePath, 'utf8').split('\n');
  for (const raw of lines) {
    const line = raw.trim();
    if (!line || line.startsWith('#')) continue;

    const ruleMatch = line.match(RULE_PATTERN_RE);
    if (ruleMatch) {
      const ruleId = ruleMatch[1];
      const glob = ruleMatch[2].trim();
      try {
        result.rulePatterns.push({ ruleId, glob, regex: globToRegex(glob) });
      } catch {
        // Skip invalid patterns
      }
    } else {
      try {
        result.filePatterns.push(globToRegex(line));
      } catch {
        // Skip invalid patterns
      }
    }
  }

  return result;
}

/**
 * Check if a file path should be ignored entirely.
 * @param {string} filePath
 * @param {{ filePatterns: RegExp[] }} patterns
 * @returns {boolean}
 */
function shouldIgnoreFile(filePath, patterns) {
  for (const regex of patterns.filePatterns) {
    if (regex.test(filePath)) return true;
  }
  return false;
}

/**
 * Check if a specific finding (rule+file) should be ignored.
 * @param {string} ruleId
 * @param {string} filePath
 * @param {{ rulePatterns: Array<{ruleId: string, regex: RegExp}> }} patterns
 * @returns {boolean}
 */
function shouldIgnoreFinding(ruleId, filePath, patterns) {
  for (const rp of patterns.rulePatterns) {
    if (rp.ruleId === ruleId && rp.regex.test(filePath)) return true;
  }
  return false;
}

/**
 * Check if a line has an inline gate-ignore comment.
 * @param {string} line
 * @returns {boolean}
 */
function hasInlineIgnore(line) {
  return INLINE_IGNORE_RE.test(line);
}

/**
 * Extract the reason from a gate-ignore comment, if any.
 * @param {string} line
 * @returns {string|null}
 */
function getIgnoreReason(line) {
  const match = line.match(INLINE_IGNORE_RE);
  if (match && match[1]) return match[1].trim();
  return null;
}

// src/cli/output.js
const RED = '\x1b[31m';
const YELLOW = '\x1b[33m';
const CYAN = '\x1b[36m';
const DIM = '\x1b[2m';
const BOLD = '\x1b[1m';
const RESET = '\x1b[0m';
const RED_BG = '\x1b[41m\x1b[37m';

const SEVERITY_COLORS = { critical: RED, high: YELLOW, medium: CYAN, low: DIM };

function shouldUseColor(setting) {
  if (process.env.NO_COLOR) return false;
  if (setting === true) return true;
  if (setting === false) return false;
  // 'auto' — check if stdout is a TTY
  return process.stdout.isTTY || false;
}

function formatSeverity(severity, useColor) {
  const label = severity.toUpperCase();
  if (!useColor) return label;
  const color = SEVERITY_COLORS[severity] || DIM;
  return `${BOLD}${color}${label}${RESET}`;
}

function formatCodeContext(lines, lineNum, contextLines, useColor) {
  const start = Math.max(0, lineNum - 1 - contextLines);
  const end = Math.min(lines.length, lineNum + contextLines);
  const gutterWidth = String(end).length;
  const out = [];

  for (let i = start; i < end; i++) {
    const num = String(i + 1).padStart(gutterWidth);
    const isFindingLine = (i === lineNum - 1);
    const prefix = isFindingLine ? `${useColor ? BOLD : ''}  ${num}   ` : `${useColor ? DIM : ''}  ${num}   `;
    const suffix = useColor ? RESET : '';
    out.push(`${prefix}${lines[i]}${suffix}`);
  }
  return out.join('\n');
}

function formatUnderline(line, matchStart, matchLength, useColor) {
  if (matchStart === undefined || matchLength === undefined) return '';
  const pad = ' '.repeat(matchStart);
  const mark = '\u203e'.repeat(matchLength); // overline character
  if (useColor) return `         ${pad}${RED}${mark}${RESET}`;
  return `         ${pad}${mark}`;
}

function formatFinding(finding, fileLines, options) {
  const useColor = options.color !== undefined ? options.color : false;
  const contextLines = options.context_lines || 2;
  const parts = [];

  // File + line header
  const loc = `${finding.file || 'unknown'}:${finding.lineNumber}`;
  parts.push(`  ${useColor ? BOLD : ''}${loc}${useColor ? RESET : ''}`);

  // Code context
  if (fileLines && fileLines.length > 0) {
    parts.push(formatCodeContext(fileLines, finding.lineNumber, contextLines, useColor));
    if (finding.matchStart !== undefined) {
      parts.push(formatUnderline(fileLines[finding.lineNumber - 1], finding.matchStart, finding.matchLength, useColor));
    }
  }

  // Rule name + severity
  const sev = formatSeverity(finding.severity || 'medium', useColor);
  parts.push(`  ${finding.ruleName || finding.ruleId} \u2500\u2500 ${sev}`);

  // Remediation one-liner (if available)
  if (finding.remediation) {
    parts.push(`  ${finding.remediation}`);
  }

  return parts.join('\n');
}

function formatSummary(counts, useColor) {
  const parts = [];
  const items = [];
  if (counts.critical > 0) items.push(`${counts.critical} critical`);
  if (counts.high > 0) items.push(`${counts.high} high`);
  if (counts.medium > 0) items.push(`${counts.medium} medium`);
  if (counts.low > 0) items.push(`${counts.low} low`);

  parts.push(`  ${useColor ? DIM : ''}\u2500`.repeat(50) + `${useColor ? RESET : ''}`);
  if (counts.total === 0) {
    parts.push('  no secrets found');
  } else {
    parts.push(`  commit blocked \u00b7 ${items.join(' \u00b7 ')}`);
    parts.push('');
    parts.push('  run gate fix to auto-remediate');
    parts.push('  run gate scan --interactive for guided walkthrough');
  }
  parts.push(`  ${useColor ? DIM : ''}\u2500`.repeat(50) + `${useColor ? RESET : ''}`);
  return parts.join('\n');
}

function formatForCI(finding, platform) {
  if (platform === 'github-actions') {
    return `::error file=${finding.file},line=${finding.lineNumber}::${finding.ruleName} (${finding.severity})`;
  }
  // Default: plain text
  return `ERROR: ${finding.file}:${finding.lineNumber} ${finding.ruleName} (${finding.severity})`;
}

function detectCI() {
  if (process.env.GITHUB_ACTIONS) return 'github-actions';
  if (process.env.GITLAB_CI) return 'gitlab-ci';
  if (process.env.CIRCLECI) return 'circleci';
  if (process.env.CI) return 'ci';
  return null;
}

module.exports = {
  shouldUseColor, formatSeverity, formatCodeContext, formatUnderline,
  formatFinding, formatSummary, formatForCI, detectCI,
  RED, YELLOW, CYAN, DIM, BOLD, RESET, RED_BG,
};
```
